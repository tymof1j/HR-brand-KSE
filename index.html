def classify_single(content: str) -> str:
    """–ö–ª–∞—Å–∏—Ñ—ñ–∫—É—î –æ–¥–∏–Ω —Ä—è–¥–æ–∫ –∫–æ–Ω—Ç–µ–Ω—Ç—É"""

    prompt = f"""–ö–ª–∞—Å–∏—Ñ—ñ–∫—É–π —Ü–µ–π –∫–æ–Ω—Ç–µ–Ω—Ç —É –û–î–ù–£ –∫–∞—Ç–µ–≥–æ—Ä—ñ—é. –í—ñ–¥–ø–æ–≤—ñ–¥—å - –¢–Ü–õ–¨–ö–ò –Ω–∞–∑–≤–∞ –∫–∞—Ç–µ–≥–æ—Ä—ñ—ó —É–∫—Ä–∞—ó–Ω—Å—å–∫–æ—é, –Ω—ñ—á–æ–≥–æ –±—ñ–ª—å—à–µ.

{CATEGORIES}

–ö–æ–Ω—Ç–µ–Ω—Ç –¥–ª—è –∫–ª–∞—Å–∏—Ñ—ñ–∫–∞—Ü—ñ—ó:
{str(content)[:1500]}

–ö–∞—Ç–µ–≥–æ—Ä—ñ—è:"""

    try:

def classify_batch(contents: list[str], batch_size: int = 5) -> list[str]:
    """
    –ö–ª–∞—Å–∏—Ñ—ñ–∫—É—î batch –∫–æ–Ω—Ç–µ–Ω—Ç—É –æ–¥–Ω–∏–º –∑–∞–ø–∏—Ç–æ–º –¥–ª—è –µ–∫–æ–Ω–æ–º—ñ—ó API calls.

    """
    if not contents:
        return []

    # –§–æ—Ä–º—É—î–º–æ batch prompt
    items_text = "\n---\n".join([
        f"[{i+1}] {str(c)[:500]}"
        for i, c in enumerate(contents)
        if c and not pd.isna(c)
    ])

    prompt = f"""–ö–ª–∞—Å–∏—Ñ—ñ–∫—É–π –∫–æ–∂–µ–Ω –µ–ª–µ–º–µ–Ω—Ç –≤ –û–î–ù–£ –∫–∞—Ç–µ–≥–æ—Ä—ñ—é. 
–§–æ—Ä–º–∞—Ç –≤—ñ–¥–ø–æ–≤—ñ–¥—ñ: –Ω–æ–º–µ—Ä - –∫–∞—Ç–µ–≥–æ—Ä—ñ—è (–ø–æ –æ–¥–Ω–æ–º—É –Ω–∞ —Ä—è–¥–æ–∫, –±–µ–∑ –ø–æ—è—Å–Ω–µ–Ω—å)

–ö–∞—Ç–µ–≥–æ—Ä—ñ—ó:
{CATEGORIES}

–ï–ª–µ–º–µ–Ω—Ç–∏:
{items_text}

–í—ñ–¥–ø–æ–≤—ñ–¥—ñ:"""

    try:
        response = model.generate_content(
            prompt,
            generation_config=genai.types.GenerationConfig(
                max_output_tokens=200,
                temperature=0.1
            )
        )

        # –ü–∞—Ä—Å–∏–Ω–≥ –≤—ñ–¥–ø–æ–≤—ñ–¥—ñ
        lines = response.text.strip().split('\n')
        results = ["–Ü–Ω—à—ñ"] * len(contents)
        valid_categories = ["–û—Å–≤—ñ—Ç–∞", "–í–ø–ª–∏–≤", "–ö–∞—Ä'—î—Ä–∞", "–ù–æ–≤–∏–Ω–∏", "–ö–æ–º'—é–Ω—ñ—Ç—ñ", "–Ü–Ω—à—ñ"]

        return results
    except Exception as e:
        print(f"Batch –ø–æ–º–∏–ª–∫–∞: {e}")
        return ["–Ü–Ω—à—ñ"] * len(contents)


def process_sheet(df: pd.DataFrame, sheet_name: str) -> pd.DataFrame:
    """–û–±—Ä–æ–±–ª—è—î –æ–¥–∏–Ω –ª–∏—Å—Ç –∑ –±–∞—Ç—á–∏–Ω–≥–æ–º"""
    if 'content' not in df.columns:
        print(f"  ‚ö†Ô∏è –õ–∏—Å—Ç '{sheet_name}' –Ω–µ –º–∞—î –∫–æ–ª–æ–Ω–∫–∏ 'content', –ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ")
        df['category'] = "–Ü–Ω—à—ñ"
        return df

    contents = df['content'].tolist()
    total = len(contents)
    print(f"  üìä –û–±—Ä–æ–±–∫–∞ {total} —Ä—è–¥–∫—ñ–≤...")

    # Batch processing - –ø–æ 5 –µ–ª–µ–º–µ–Ω—Ç—ñ–≤ –∑–∞ –∑–∞–ø–∏—Ç
    batch_size = 5
    all_results = []

    for i in range(0, total, batch_size):
        batch = contents[i:i+batch_size]
        batch_num = i // batch_size + 1
        total_batches = (total + batch_size - 1) // batch_size

        print(f"    Batch {batch_num}/{total_batches}...", end=" ")

        results = classify_batch(batch, batch_size)
        all_results.extend(results)

        print(f"‚úì")

        # Rate limiting: ~4 —Å–µ–∫—É–Ω–¥–∏ –º—ñ–∂ –∑–∞–ø–∏—Ç–∞–º–∏ –¥–ª—è 15 RPM
        if i + batch_size < total:
            time.sleep(4)

    df['category'] = all_results[:total]
    return df
